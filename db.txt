create or replace function public.get_customer_list_data(
  p_query text default null,
  p_start_date date default null,
  p_end_date date default null,
  p_min_total numeric default null,
  p_max_total numeric default null,
  p_min_claimed numeric default null,
  p_max_claimed numeric default null,
  p_min_unclaimed numeric default null,
  p_max_unclaimed numeric default null,
  p_has_claimed boolean default false,
  p_has_eligible_claims boolean default false,
  p_page integer default 1,
  p_items_per_page integer default 10
)
returns jsonb
language sql
security definer
set search_path = public
as $$
with filtered as (
  select
    "CUSTOMER CODE" as customer_code,
    "CUSTOMER NAME" as customer_name,
    "HOUSE NAME" as house_name,
    "STREET" as street,
    "PLACE" as place,
    "PIN CODE" as pin_code,
    "MOBILE" as mobile,
    "NET WEIGHT" as net_weight,
    original_date,
    parsed_date,
    total_points,
    claimed_points,
    unclaimed_points,
    points_last_updated
  from public.customer_summary
  where
    (
      coalesce(p_query, '') = ''
      or "CUSTOMER CODE" ilike '%' || p_query || '%'
      or "CUSTOMER NAME" ilike '%' || p_query || '%'
      or "MOBILE" ilike '%' || p_query || '%'
    )
    and (p_start_date is null or parsed_date >= p_start_date)
    and (p_end_date is null or parsed_date <= p_end_date)
    and (p_min_total is null or total_points >= p_min_total)
    and (p_max_total is null or total_points <= p_max_total)
    and (p_min_claimed is null or claimed_points >= p_min_claimed)
    and (p_max_claimed is null or claimed_points <= p_max_claimed)
    and (p_min_unclaimed is null or unclaimed_points >= p_min_unclaimed)
    and (p_max_unclaimed is null or unclaimed_points <= p_max_unclaimed)
    and (not p_has_claimed or claimed_points > 0)
    and (not p_has_eligible_claims or unclaimed_points >= 5)
),
stats as (
  select
    count(*)::int as total_count,
    count(*) filter (where unclaimed_points >= 5)::int as eligible_count,
    coalesce(sum(total_points), 0) as total_points,
    coalesce(sum(claimed_points), 0) as total_claimed,
    coalesce(sum(unclaimed_points), 0) as total_unclaimed
  from filtered
),
paged as (
  select *
  from filtered
  order by customer_code asc
  offset greatest((coalesce(p_page, 1) - 1) * coalesce(p_items_per_page, 10), 0)
  limit greatest(coalesce(p_items_per_page, 10), 1)
)
select jsonb_build_object(
  'rows', coalesce((select jsonb_agg(to_jsonb(paged)) from paged), '[]'::jsonb),
  'total_count', stats.total_count,
  'eligible_count', stats.eligible_count,
  'total_points', stats.total_points,
  'total_claimed', stats.total_claimed,
  'total_unclaimed', stats.total_unclaimed
)
from stats;
$$;

grant execute on function public.get_customer_list_data(
  text, date, date, numeric, numeric, numeric, numeric, numeric, numeric, boolean, boolean, integer, integer
) to anon, authenticated;
